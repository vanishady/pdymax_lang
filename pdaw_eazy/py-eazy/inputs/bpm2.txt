patch bpm2

import pdaw1
import pdaw2

#start/stop audio stream
@onoff()

#set bpm
@bpm(140)

#instruments
drumseq = @drums([0,0,0,1,0,0,0,1,0,0,0,1,0,0,0,1], '../samples/kick.wav', 'kick')

#clock
drumclock = @clockdiv(16)

#connect intrument to clock
@bind(drumclock,drumseq,16)

#mixer
mix = @basicmixer(['kick'], [0.4])
